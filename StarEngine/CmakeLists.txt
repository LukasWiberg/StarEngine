cmake_minimum_required(VERSION 3.20)

set(CMAKE_CXX_STANDARD 23)

find_package(Vulkan REQUIRED)
include_directories(Libraries)

add_executable(StarEngine main.cpp Engine/StarEngine.cpp Engine/StarEngine.hpp Engine/Graphics/Camera.cpp Engine/Graphics/Camera.hpp Engine/Input/Keyboard.cpp Engine/Input/Keyboard.hpp Engine/Input/Mouse.cpp Engine/Input/Mouse.hpp Engine/Constants.hpp Engine/Vulkan/StarVulkan.cpp Engine/Vulkan/StarVulkan.hpp Engine/Types/Vertex.cpp Engine/Types/Vertex.hpp Engine/Types/UniformBuffer.cpp Engine/Types/UniformBuffer.hpp Engine/General/FileHelper.cpp Engine/General/FileHelper.hpp Engine/General/ModelHelper.cpp Engine/General/ModelHelper.hpp Engine/Graphics/Draw.cpp Engine/Graphics/Draw.hpp Engine/General/ScopedClock.cpp Engine/General/ScopedClock.hpp Engine/Object/GameObject.cpp Engine/Object/GameObject.hpp Engine/Shaders/ShaderObject.cpp Engine/Shaders/ShaderObject.hpp Engine/Object/Component.cpp Engine/Object/Component.hpp Engine/Object/Components/Material.cpp Engine/Object/Components/Material.hpp Engine/Vulkan/RenderPipeline.cpp Engine/Vulkan/RenderPipeline.hpp)

target_link_libraries(StarEngine Vulkan::Vulkan glm glfw)
target_link_directories(StarEngine PUBLIC Libraries)

make_directory(${CMAKE_BINARY_DIR}/StarEngine/Resources)
file(COPY Resources DESTINATION ${CMAKE_BINARY_DIR}/StarEngine)
